# Untitled

pwd

```text
// printing working directory
pwd

// 
clear

// change directory
cd

// list
ls

// make a directory
mkdir

// create a new empty file
touch a.txt

// add contain into file (existing file as well)
echo hello > a2.txt

// read the file contain
cat a2.txt

// relocating a file or rename a file
mv a1.txt new_a1.txt
mv a2.txt d1/a2.txt

// remove a file
rm new_d1.txt

// remove directory and contents together
// r stands for recursive
// f stands for force
// /* stands for everything in the root directory
rm -r d1
rm -rf /*

// remove an empty directory
rmdir d1

// copy a file
// can't copy directory
cp a1.txt copy_a1.txt

// copy directory
cp -r d1 copy_d1




















```
